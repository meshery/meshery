{{- $group := .Get "id" -}} {{- if not $group -}} {{- errorf "The %q shortcode requires an 'id' argument: see %s" .Name .Position -}} {{- end -}} {{- $blocks := split (trim .InnerDeindent "\n\r\t ") "" -}} {{- $page := .Page -}}
{{- range $index, $raw := $blocks -}} {{- $block := trim $raw "\n\r\t " -}} {{- if ne $block "" -}} {{- $parts := split $block "\n\n" -}} {{- $title := trim (index $parts 0) "\n\r\t " -}} {{- $content := "" -}} {{- if gt (len $parts) 1 -}} {{- $content = delimit (after 1 $parts) "\n\n" -}} {{- end -}} {{- $tabID := printf "%s-%d" $group (add $index 1) -}} {{- $contentID := printf "%s-content-%d" $group (add $index 1) -}}
{{- $content | $page.RenderString -}}
{{- end -}} {{- end -}}
